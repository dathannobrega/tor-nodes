import requests

# Buscar todos os nós
response = requests.get('https://tor.protexion.cloud/api/nodes')
nodes = response.json()

# Filtrar apenas nós exit ativos
exit_nodes = [node for node in nodes['nodes']
              if node['exit_node'] and node['running']]

print(f"Nós exit ativos: {len(exit_nodes)}")

<?php
$url = 'https://tor.protexion.cloud/tornodes-ip.txt';

$response = @file_get_contents($url);
if ($response === false) {
    die("Erro ao buscar o arquivo.");
}

$lines = preg_split("/\r\n|\n|\r/", $response);

$ips = [];
foreach ($lines as $line) {
    $line = trim($line);
    // pula comentários e linhas em branco
    if ($line === '' || strpos($line, '#') === 0) {
        continue;
    }
    // opcional: valida sintaxe de IP
    if (filter_var($line, FILTER_VALIDATE_IP)) {
        $ips[] = $line;
    }
}
$total = count($ips);
echo "Total de nós Tor: {$total}\n";
?>
